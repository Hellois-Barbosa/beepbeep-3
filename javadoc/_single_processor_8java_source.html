194  }
195  // Check if each input queue has an event ready
196  for (int i = 0; i < m_inputArity; i++)
197  {
198  Queue<Object> queue = m_inputQueues[i];

200  {
201  // One of them doesn't: we can't produce an output yet
202  return this;
203  }
204  }

206  Object[] inputs = new Object[m_inputArity];
207  for (int i = 0; i < m_inputArity; i++)
208  {

382  {
383  Queue<Object> out_queue = m_outputQueues[m_index];
384  // If an event is already waiting in the output queue,
385  // return it and don't pull anything from the input
386  if (!out_queue.isEmpty())
387  {
388  return true;
389  }
390  // Check if each pullable has an event ready
391  for (int tries = 0; tries < Processor.MAX_PULL_RETRIES; tries++)
392  {
393  for (int i = 0; i < m_inputArity; i++)
394  {
395  Pullable p = m_inputPullables[i];
396  if (p == null)
397  {
398  throw new PullableException("Input " + i + " of this processor is connected to nothing",
399  getProcessor());
400  }
401  boolean status = p.hasNext();
402  if (!status)
403  {
404  return false;
405  }
406  }
407  // We are here only if every input pullable has answered YES
408  // Pull an event from each
409  Object[] inputs = new Object[m_inputArity];
410  for (int i = 0; i < m_inputArity; i++)
411  {
412  Pullable p = m_inputPullables[i];
413  // Don't check for p == null, we did it above
414  Object o = p.pull();
415  inputs[i] = o;
416  }
417  // Compute output event(s)
418  m_tempQueue.clear();
419  boolean computed;
420  try
421  {
422  computed = compute(inputs, m_tempQueue);
423  }
424  catch (ProcessorException e)
425  {
426  throw new PullableException(e);
427  }
428  NextStatus status_to_return = NextStatus.NO;
429  if (!computed)
430  {
431  // No output will ever be returned: stop there
432  return false;
433  }
434  if (!m_tempQueue.isEmpty())
435  {
436  // We computed an output event; add it to the output queue
437  // and answer YES
438  for (Object[] evt : m_tempQueue)
439  {
440  if (evt != null)
441  {
442  for (int i = 0; i < m_outputArity; i++)
443  {
444  Queue<Object> queue = m_outputQueues[i];
445  queue.add(evt[i]);
446  }
447  status_to_return = NextStatus.YES;
448  }
449  else
450  {
451  // This source will NEVER output anything again
452  return false;
453  }
454  }
455  if (status_to_return == NextStatus.YES)
456  {
457  return true;
458  }
459  }
460  // Otherwise, try the whole thing again
461  }
462  return false;

465  @Override
466  public synchronized NextStatus hasNextSoft()
467  {
468  Queue<Object> out_queue = m_outputQueues[m_index];
469  // If an event is already waiting in the output queue,
470  // return yes and don't pull anything from the input
471  if (!out_queue.isEmpty())
472  {
473  return NextStatus.YES;
474  }
475  // Check if each pullable has an event ready
476  for (int i = 0; i < m_inputArity; i++)
477  {
478  Pullable p = m_inputPullables[i];
479  NextStatus status = p.hasNextSoft();
480  if (status == NextStatus.NO)
481  {
482  return NextStatus.NO;
483  }
484  if (status == NextStatus.MAYBE)
485  {
486  return NextStatus.MAYBE;
487  }
488  }
489  // We are here only if every input pullable has answered YES
490  // Pull an event from each
491  Object[] inputs = new Object[m_inputArity];
492  {
493  int i = 0;
494  for (Pullable p : m_inputPullables)
495  {
496  inputs[i] = p.pullSoft();
497  i++;
498  }
499  }
500  // Compute output event(s)
501  NextStatus status_to_return = NextStatus.MAYBE;
502  m_tempQueue.clear();
503  boolean computed;
504  try
505  {
506  computed = compute(inputs, m_tempQueue);
507  }
508  catch (ProcessorException e)
509  {
510  throw new PullableException(e);
511  }
512  if (!computed)
513  {
514  status_to_return = NextStatus.NO;
515  }
516  else if (!m_tempQueue.isEmpty())
517  {
518  for (Object[] evt : m_tempQueue)
519  {
520  if (evt != null)
521  {
522  // We computed an output event; add it to the output queue
523  // and answer YES
524  int i = 0;
525  for (Queue<Object> queue : m_outputQueues)
526  {
527  queue.add(evt[i]);
528  i++;
529  }
530  status_to_return = NextStatus.YES;
531  }
532  }
533  }
534  return status_to_return;
535  }

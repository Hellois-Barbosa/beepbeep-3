159  public static boolean parseBoolValue(Object o)
160  {
161  if (o instanceof Boolean)
162  {
163  return (Boolean) o;
164  }
165  else if (o instanceof String)
166  {
167  String s = (String) o;
168  return s.compareToIgnoreCase("true") == 0 || s.compareToIgnoreCase("T") == 0
169  || s.compareToIgnoreCase("1") == 0;
170  }
171  if (o instanceof Number)
172  {
173  Number n = (Number) o;
174  return Math.abs(n.doubleValue()) >= 0.00001;
175  }
176  // When in doubt, return false

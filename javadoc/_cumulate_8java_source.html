42  protected boolean compute(Object[] inputs, Object[] outputs)
43  {
44  // We override compute() from FunctionProcessor, only to complete the
45  // association between input and output events (each output event depends on
46  // the current input event and also the previous output event front)
47  boolean b = super.compute(inputs, outputs);
48  if (m_eventTracker != null)
49  {
50  for (int j = 0; j < outputs.length; j++)
51  {
52  if (m_outputCount > 1)
53  {
54  for (int k = 0; k < outputs.length; k++)
55  {
56  // The output event is associated to the current input event
57  // (this is already taken care of by super.compute)...
58  // ...and to the previous output event
59  // -1 and -2 since the count has already been incremented by the
60  // call to super.compute() above
61  associateToOutput(j, m_outputCount - 2, k, m_outputCount - 1);
62 
63  }
64  }
65  else
66  {
67  for (int k = 0; k < outputs.length; k++)
68  {
69  associateTo(new StartValue(getId(), j), k, m_outputCount - 1);
70  }
71  }
72  }
73  }
74  return b;
75  }

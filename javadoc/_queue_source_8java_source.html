140  @Override
141  @SuppressWarnings("squid:S1168")
142  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
143  {
144  int size = m_events.size();
145  if (m_index >= size)
146  {
147  return false;
148  }
149  Object[] output = new Object[getOutputArity()];
150  Object event = m_events.get(m_index);
151  if (m_loop)
152  {
153  m_index = (m_index + 1) % size;
154  }
155  else
156  {
157  // If we don't loop, play the events only once
158  m_index++;
159  }
160  if (m_index > size && !m_loop)
161  {
162  // No more events from this queue
163  return false;
164  }
165  for (int i = 0; i < getOutputArity(); i++)
166  {
167  if (event == null)
168  {
169  // If one of the elements is null, don't output anything
170  return true;
171  }
172  output[i] = event;
173  }
174  outputs.add(output);
175  if (m_eventTracker != null)
176  {
177  int index = m_index - 1;
178  if (index < 0)
179  {
180  index += size;
181  }
182  for (int i = 0; i < getOutputArity(); i++)
183  {
184  associateTo(new QueueFunction(getId(), index), i, m_outputCount);
185  }
186  }
187  m_outputCount++;
188  return true;
189  }

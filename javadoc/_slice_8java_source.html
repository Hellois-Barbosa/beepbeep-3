132  @Override
133  protected boolean compute(Object[] inputs, Object[] outputs)
134  {
135  int output_arity = getOutputArity();
136  Object[] f_value = new Object[1];
137  try
138  {
139  m_slicingFunction.evaluate(inputs, f_value);
140  }
141  catch (FunctionException e)
142  {
143  throw new ProcessorException(e);
144  }
145  Object slice_ids = f_value[0];
146  if (slice_ids == null)
147  {
148  // This event applies to no slice; don't bother processing it
149  outputs[0] = m_lastValues;
150  return true;
151  }
152  Object[] slice_vals;
153  if (m_explodeArrays)
154  {
155  if (slice_ids.getClass().isArray())
156  {
157  slice_vals = (Object[]) slice_ids;
158  }
159  if (slice_ids instanceof Collection)
160  {
161  Collection<?> col = (Collection<?>) slice_ids;
162  slice_vals = new Object[col.size()];
163  int i = 0;
164  for (Object o : col)
165  {
166  slice_vals[i] = o;
167  i++;
168  }
169  }
170  else
171  {
172  slice_vals = new Object[] { slice_ids };
173  }
174  }
175  else
176  {
177  slice_vals = new Object[] { slice_ids };
178  }
179  for (Object slice_id : slice_vals)
180  {
181  Set<Object> slices_to_process = new HashSet<Object>();
182  if (slice_id instanceof ToAllSlices || slice_id == null)
183  {
184  slices_to_process.addAll(m_slices.keySet());
185  }
186  else
187  {
188  if (!m_slices.containsKey(slice_id))
189  {
190  // First time we see this value: create new slice
191  Processor p = m_processor.duplicate();
192  m_slices.put(slice_id, p);
193  addContextFromSlice(p, slice_id);
194  QueueSink sink = new QueueSink(output_arity);
195  Connector.connect(p, sink);
196  m_sinks.put(slice_id, sink);
197  // Put dummy value temporarily
198  m_lastValues.put(slice_id, null);
199  }
200  slices_to_process.add(slice_id);
201  }
202  @SuppressWarnings("unchecked")
203  Future<Pushable>[] futures = new Future[slices_to_process.size()];
204  for (Object s_id : slices_to_process)
205  {
206  // Find processor corresponding to that slice
207  Processor slice_p = m_slices.get(s_id);
208  // If this slice hasn't been cleaned up...
209  if (slice_p != null)
210  {
211  QueueSink sink_p = m_sinks.get(s_id);
212  // Push the input into the processor
213  // Pushable[] p_array = new Pushable[inputs.length];
214  for (int i = 0; i < inputs.length; i++)
215  {
216  Object o_i = inputs[i];
217  Pushable p = slice_p.getPushableInput(i);
218  Future<Pushable> f = p.pushFast(o_i);
219  if (f == null)
220  {
221  System.out.println("NULL");
222  }
223  futures[i] = f;
224  // p_array[i] = p;
225  }
226  // Wait for all slices to be done
227  for (Future<Pushable> f : futures)
228  {
229  if (f != null)
230  {
231  try
232  {
233  f.get();
234  }
235  catch (InterruptedException e)
236  {
237  throw new ProcessorException(e);
238  }
239  catch (ExecutionException e)
240  {
241  throw new ProcessorException(e);
242  }
243  }
244  }
245  // Collect the output from that processor
246  Object[] out = sink_p.remove();
247  // Can we clean that slice?
248  Object[] can_clean = new Object[1];
249  if (m_cleaningFunction != null)
250  {
251  try
252  {
253  m_cleaningFunction.evaluate(out, can_clean);
254  }
255  catch (FunctionException e)
256  {
257  throw new ProcessorException(e);
258  }
259  }
260  if (can_clean != null && can_clean.length > 0 && can_clean[0] instanceof Boolean
261  && (Boolean) (can_clean[0]))
262  {
263  // Yes: remove the processor for that slice
264  m_slices.remove(s_id);
265  m_sinks.remove(s_id);
266  }
267  m_lastValues.put(s_id, out[0]);
268  }
269  }
270  outputs[0] = m_lastValues;
271  }
272  return true;
273  }
